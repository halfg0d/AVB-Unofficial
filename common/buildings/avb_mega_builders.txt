# mb = MegaBuilding
# every building here is equally to last version of vanilla
# positive modifers are equall to previous tier
# cost is sum of version until it
# upkeep 50% higherk
@avb_mb_time = 90
building_avb_mb_lab = {
	base_buildtime = @avb_mb_time
	category = research
	icon = building_research_lab_3
	prerequisites = {
		"tech_avb_mb_lab"
	}
	potential = {
		avb_hg_changes = yes
		NOR = {
			has_modifier = resort_colony
			has_modifier = slave_colony
		}
	}
	allow = {
		has_major_upgraded_capital = yes
	}
	destroy_trigger = {
		OR = {
			has_modifier = resort_colony
			has_modifier = slave_colony
		}
	}
	inline_script = {
		script = jobs/researchers_add
		AMOUNT = @b2_jobs
	}
	resources = {
		category = planet_buildings
		cost = {
			minerals = @[b1_minerals+b2_minerals+b3_minerals]
			exotic_gases = @[b2_rare_cost+b3_rare_cost]
		}
		upkeep = {
			energy = @[b3_upkeep * 1.5]
			exotic_gases = @[b3_rare_upkeep * 1.5]
		}
	}
	upgrades = {
		building_avb_research_lab_4
	}
}
building_avb_mb_commercial = {
	base_buildtime = @avb_mb_time
	category = trade
	icon = building_commercial_megaplex
	prerequisites = {
		"tech_avb_mb_commercial"
	}
	potential = {
		avb_hg_changes = yes
		exists = owner
		owner = { is_regular_empire = yes }
		NOT = { has_modifier = slave_colony }
		NOT = { has_building = building_low_tech_capital }
	}
	destroy_trigger = {
		exists = owner
		OR = {
			owner = { is_regular_empire = no }
			has_modifier = slave_colony
		}
	}
	resources = {
		category = planet_buildings
		cost = {
			minerals = @[b1_minerals+b2_minerals]
			rare_crystals = @b2_rare_cost
		}
		upkeep = {
			energy = @[b2_upkeep * 1.5]
			rare_crystals = @[b2_rare_upkeep * 1.5]
		}
	}
	upgrades = {
		building_avb_commercial_3
	}
	planet_modifier = {
		job_clerk_add = 2
		job_trader_add = 1
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				has_active_tradition = tr_mercantile_commercial_enterprise
			}
		}
		modifier = {
			job_trader_add = 1
		}
	}

	triggered_desc = {
		text = job_clerk_effect_desc
	}

	triggered_desc = {
		text = job_trader_effect_desc
	}
	show_tech_unlock_if = {
		is_regular_empire = yes
	}
}

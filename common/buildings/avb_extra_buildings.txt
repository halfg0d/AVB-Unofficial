#Mining Buildings
building_avb_mineral_mine_1 = {
	base_buildtime = @b1_time

	category = resource

	allow = { always = yes }

	potential = {
		always = no
		avb_hg_changes = yes
		NOR = {
			has_modifier = resort_colony
			has_modifier = crucible_colony
		}

		NOR = { # Ringworlds don't have mines
			is_planet_class = pc_ringworld_habitable
			is_planet_class = pc_cybrex
		}
		OR = {
			NOT = { is_planet_class = pc_habitat }
			AND = {
				is_planet_class = pc_habitat
				check_modifier_value = {
					modifier = district_hab_mining_max
					value > 0
				}
			}
		}
		OR = {
			owner = { is_ai = no }
			AND = {
				owner = {
					is_ai = yes
					has_monthly_income = {
						resource = minerals
						value < 10
					}
				}
				free_district_slots = 0
			}
		}
	}
	destroy_trigger = {
		exists = owner
		OR = {
			planet = { 
				has_modifier = resort_colony
			}
			planet = {
				free_building_slots = 0
				owner = {
					is_ai = yes
					has_origin = origin_subterranean
					has_monthly_income = {
						resource = minerals
						value > 50
					}
				}
			}
			planet = {
				free_building_slots = 0
				owner = {
					is_ai = yes
					NOT = { has_origin = origin_subterranean }
					has_monthly_income = {
						resource = minerals
						value > 100
					}
				}
			}
		}
	}

	inline_script = {
		script = jobs/miners_add
		AMOUNT = 2
	}

	show_tech_unlock_if = {
		always = no
		avb_hg_changes = yes
	}
	resources = {
		category = planet_buildings
		cost = {
			minerals = @b1_minerals
		}
		upkeep = {
			energy = @b1_upkeep
		}
		produces = {
			minerals = 10
		}
	}

	prerequisites = {
		"tech_mining_2"
	}
	
	upgrades = {
		building_avb_mineral_mine_2
	}
	ai_resource_production = {
		minerals = 5
		trigger = {
			always = yes
		}
	}
}

building_avb_core_miner = {
	base_buildtime = @avb_b3_time
	base_cap_amount = 1
	category = resource
	allow = {
		has_major_upgraded_capital = yes
	}
	potential = {
		always = no
		avb_hg_changes = yes
		is_artificial = no
		OR = {
			owner = { is_ai = no }
			AND = {
				owner = { is_ai = yes }
				has_any_mining_district = yes
			}
		}
	}
	destroy_trigger = {
		OR = {
			planet = {
				is_artificial = yes
			}
			AND = {
				owner = { is_ai = yes }
				has_any_mining_district = no
			}
		}
	}
	planet_modifier = {
		custom_tooltip = avb_core_miner_tooltip
		planet_miners_minerals_produces_add = 1
		planet_miners_energy_produces_add = 0.5
		planet_miners_alloys_produces_add = 0.25
	}
	triggered_planet_modifier = {
		potential = {
			planet = {
				has_designation = col_mining
			}
		}
		modifier = {
			planet_miners_minerals_produces_add = 0.5
		}

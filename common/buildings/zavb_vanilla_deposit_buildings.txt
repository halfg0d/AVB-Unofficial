building_xeno_zoo = {
	base_buildtime = @b1_time
	base_cap_amount = 1
	category = amenity
	
	potential = {
		exists = owner
		owner = {
			is_gestalt = no
		}
		planet = {
			OR = {
				has_deposit = d_alien_pets_deposit
				AND = {
					exists = owner
					owner = { has_policy_flag = pre_sapients_protect }
					any_owned_pop = {
						is_sapient = no
					}
				}
				any_owned_pop = {
					is_livestock = yes
				}
			}
		}
	}

	destroy_trigger = {
		OR = {
			AND = {
				exists = owner
				owner = {
					is_regular_empire = no
				}
			}
			planet = {
				NOR = {
					has_deposit = d_alien_pets_deposit
					AND = {
						exists = owner
						owner = { has_policy_flag = pre_sapients_protect }
						any_owned_pop = {
							is_sapient = no
						}
					}
					any_owned_pop = {
						is_livestock = yes
					}
				}
			}			
		}
	}

	resources = {
		category = planet_buildings
		cost = {
			minerals = @b1_minerals
		}
		upkeep = {
			energy = @b1_upkeep
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				OR = {
					has_valid_civic = civic_warrior_culture
				}
			}
		}
		modifier = {
			job_duelist_add = 1
			job_entertainer_add = -1
		}
	}

	planet_modifier = {
		job_entertainer_add = 1
		job_xeno_zoo_keeper_add = 1
	}
	triggered_planet_modifier = {
		potential = {
			planet = {
				any_owned_pop = {
					is_livestock = yes
				}
			}
			exists = owner
			owner = {
				NOT = { has_valid_civic = civic_warrior_culture }
			}
		}
		modifier = {
			job_livestock_zoo_animal_add = 3
		}
	}

	triggered_planet_modifier = {
		potential = {
			planet = {
				any_owned_pop = {
					is_livestock = no
				}
			}
			exists = owner
			owner = {
				has_valid_civic = civic_warrior_culture
			}
		}
		modifier = {
			job_livestock_zoo_beast_add = 3
		}
	}

	prerequisites = {
		"tech_alien_life_studies"
	}

	triggered_desc = {
		text = job_xeno_zoo_keeper_effect_desc
	}

	triggered_desc = {
		trigger = {
			exists = owner
			owner = {
				NOT = { has_valid_civic = civic_warrior_culture }
			}
		}
		text = job_entertainer_effect_desc
	}

	triggered_desc = {
		trigger = {
			exists = owner
			owner = { 
				has_valid_civic = civic_warrior_culture
			 }
		}
		text = job_duelist_effect_desc
	}
	inline_script = misc/zoo_animals
	upgrades = {
		"building_avb_xeno_zoo_2"

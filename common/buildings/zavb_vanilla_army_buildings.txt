building_military_academy = {
	base_buildtime = @b2_time
	base_cap_amount = 1
	can_be_ruined = no
	category = army
	
	potential = {
		exists = owner
		owner = {
			NOT = { has_valid_civic = civic_reanimated_armies }
			OR = {
				NOT = { has_global_flg = bug_branch_active }
				AND = {
					has_global_flag = bug_branch_active
					is_bugged_reanimator = no #EC: Bug <<
				}
			}
			OR = {
				has_z_expanded_mod = no
				AND = {
					has_z_expanded_mod = yes
					NOT = { has_valid_civic = civic_machine_terminator }
				}
			}
		}
		OR = {
			has_z_expanded_mod = no
			AND = {
				has_z_expanded_mod = yes
				NOT = { has_planet_flag = planetary_station }
			}
		}
	}

	allow = {
		has_upgraded_capital = yes
		if = {
			limit = {
				exists = owner
				owner = { is_ai = yes }
			}
			OR = {
				has_building = building_stronghold
				has_building = building_fortress
				is_capital = yes
				solar_system = { is_bottleneck_system = yes }
			}
		}
	}

	destroy_trigger = {
		exists = owner
		owner = {
			OR = {
				has_valid_civic = civic_reanimated_armies
				AND = {
					is_ai = yes
					prev = {
						NOR = {
							has_building = building_stronghold
							has_building = building_fortress
							is_capital = yes
							solar_system = { is_bottleneck_system = yes }
						}
					}
				}
				AND = {
					has_z_expanded_mod = yes
					has_valid_civic = civic_machine_terminator
				}
			}
		}
	}

	convert_to = {
		building_dread_encampment
		building_war_factory #zExpanded
	}

	army_modifier = {
		army_starting_experience_add = 100
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_gestalt = no }
		}
		modifier = {
			job_soldier_add = @b1_jobs
		}
	}
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_gestalt = yes }
		}
		modifier = {
			job_warrior_drone_add = @b1_jobs
		}
	}
	#zExpanded \/
	triggered_planet_modifier = {
		potential = {
			has_z_expanded_mod = yes
		}
		modifier = {
			planet_army_build_speed_mult = 0.5
		}
	}
	triggered_country_modifier = {
		potential = {
			has_z_expanded_mod = yes
			exists = army_leader
		}
		modifier = {
			empire_size_colonies_mult = 0.1
		}
	}
	#zExpanded /\
	#EC: Classic \/
	triggered_planet_modifier = {
		potential = {
			has_global_flag = ethic_civic_classic_active
			exists = owner
			owner = { has_civic = civic_martial_brotherhood }
		}
		modifier = {
			planet_housing_add = 2
		}
	}
	#EC: Classic /\
	resources = {
		category = planet_buildings
		cost = {
			minerals = @b1_minerals
		}
		upkeep = {
			energy = @b1_upkeep
		}
		produces = {
			trigger = {
				owner = {
					has_global_flag = ethic_civic_classic_active
					has_civic = civic_martial_brotherhood
				}
			}
			unity = 5
		}
	}
	
	prerequisites = {
		"tech_centralized_command"
	}

	show_tech_unlock_if = {
		NOT = { has_valid_civic = civic_reanimated_armies }
		NOT = { has_valid_civic = civic_machine_terminator }
	}

	triggered_desc = {
		trigger = {
			exists = owner
			owner = { is_gestalt = yes }
		}
		text = job_warrior_drone_effect_desc
	}
	triggered_desc = {
		trigger = {
			exists = owner
			owner = { is_gestalt = no }
		}
		text = job_soldier_effect_desc
	}
	triggered_desc = { #zExpanded
		trigger = {
			has_z_expanded_mod = yes
		}
		text = building_military_academy_tooltip
	}
	upgrades = {
		"building_avb_military_academy_2"
		"building_eutab_battle_praxeum"
	}
}

building_fortress = {
	base_buildtime = @b1_time
	can_build = no
	planetary_ftl_inhibitor = yes
	position_priority = 100
	can_be_ruined = no
	category = army
	
	allow = {
		has_upgraded_capital = yes
	}

	destroy_trigger = {
		has_plentiful_tradition_mod = yes
		exists = owner
		owner = { is_ai = yes }
		OR = {
			AND = {
				num_buildings = { type = building_fortress value > 1 }
				free_building_slots = 0
				owner = {
					is_militarist = no
					used_naval_capacity_percent < 0.4
				}
			}
			AND = {
				num_buildings = { type = building_fortress value > 2 }
				free_building_slots = 0
				owner = {
					is_militarist = yes
					used_naval_capacity_percent < 0.4
				}
			}
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { 
				is_gestalt = no 
				NOT = {
					has_civic = civic_private_militaries
				}
			}
		}
		modifier = {
			job_soldier_add = @b2_jobs
		}
	}

building_planetary_shield_generator = {
	base_buildtime = @b3_time	
	base_cap_amount = 1
	position_priority = 100
	can_be_ruined = no
	#icon = building_planetary_shield_generator

	category = army
	
	allow = {
		has_upgraded_capital = yes
		hidden_trigger = { owner = { is_ai = no } }
		if = {
			limit = {
				exists = orbital_defence
			}
			orbital_defence = {
				exists = starbase
				starbase = {
					NOR = {
						has_starbase_building = ring_orbital_shield_generator
						is_starbase_building_building = ring_orbital_shield_generator
					}
				}
			}
		}
	}
	
	
	destroy_trigger = {
		exists = owner
		owner = { is_ai = yes }
		is_capital = no
		solar_system = { is_bottleneck_system = no }
		num_buildings = { type = building_stronghold value < 1 }
		num_buildings = { type = building_fortress value < 1 }
	}
	
	planet_modifier = {
		planet_orbital_bombardment_damage = -0.5
		}

	resources = {
		category = planet_buildings
		cost = {
			minerals = @b2_minerals
		}
		upkeep = {
			energy = @b2_upkeep
		}
		produces = {
			trigger = {
				owner = {
					has_global_flag = ethic_civic_classic_active
					has_civic = civic_inwards_perfection
				}
			}
			unity = 10
		}
	}
	#EC: Bug \/
	triggered_planet_modifier = {
		potential = {
			has_global_flag = bug_branch_active
			exists = owner
			owner = { has_civic = civic_inwards_perfection }
		}
		modifier = {
			planet_jobs_unity_produces_mult = 0.2
		}
	}
	#EC: Bug /\
	prerequisites = {
		"tech_planetary_shield_generator"
	}
	
	upgrades = {
		building_avb_planetary_shield_generator_2
	}
}

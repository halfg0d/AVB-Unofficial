building_avb_foundry_4 = {
	base_buildtime = @avb_b3_time
	can_build = no

	category = manufacturing
	
	allow = {
		has_fully_upgraded_capital = yes
	}
	
	destroy_trigger = {
		OR = {

				exists = owner
				owner = {
					is_ai = yes
				}
				has_any_industry_district = no
			}
		}
	}

	resources = {
		category = planet_buildings

			minerals = @avb_b3_minerals
			volatile_motes = @avb_b3_rare_cost
		}
		upkeep = {
			energy = @avb_b3_upkeep
			volatile_motes = @avb_b3_rare_upkeep
		}
		cost = {
			trigger = {
				has_ringworld_output_boost = yes
			}
			minerals = 700
			volatile_motes = 500
		}
		upkeep = {
			trigger = {
				has_ringworld_output_boost = yes
			}
			energy = 7
			volatile_motes = 4
		}
	}

	prerequisites = {
		"tech_avb_alloys_3"
	}	

	upgrades = {
		building_avb_foundry_5
	}
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				is_catalytic_empire = no
			}
		}
		modifier = {
			planet_metallurgists_alloys_produces_add = 3
			planet_metallurgists_minerals_upkeep_add = 6
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				is_catalytic_empire = yes
			}
		}
		modifier = {
			planet_metallurgists_alloys_produces_add = 3.75
			planet_metallurgists_food_upkeep_add = 9
		}
	}
	#Ringworld bonus
	triggered_planet_modifier = {
		potential = {
			has_ringworld_output_boost = yes
			exists = owner
			owner = {
				is_catalytic_empire = no
			}
		}
		modifier = {
			planet_metallurgists_alloys_produces_add = 1
			planet_metallurgists_minerals_upkeep_add = 2
		}
	}

	triggered_planet_modifier = {
		potential = {
			has_ringworld_output_boost = yes
			exists = owner
			owner = {
				is_catalytic_empire = yes
			}
		}


tech_ring_avb_robot_assembly_1 = {
	icon = tech_ring_avb_robot_assembly
	cost = @orb_tier4cost1
	area = engineering
	tier = 4
	category = { industry }
	prerequisites = { "tech_orbital_ring_tier_1" }
	weight = @tier4weight1

	potential = {
		is_hive_empire = no
		OR = {
			is_machine_empire = yes
			AND = {
				is_gestalt = no
				has_technology = tech_robotic_workers
			}
		}
	}	
	
	weight_modifier = {
		modifier = {
			factor = 0
			has_policy_flag = robots_outlawed
		}	
		modifier = {
			factor = 0.5
			OR = {
				has_ethic = "ethic_spiritualist"
				has_ethic = "ethic_fanatic_spiritualist"
			}
		}
		modifier = {
			factor = 2
			OR = {
				has_ethic = ethic_materialist
				has_ethic = ethic_fanatic_materialist
				is_machine_empire = yes
			}
		}
		modifier = {
			factor = 2
			OR = {
				has_origin = origin_mechanists
				has_origin = origin_machine
			}
		}
		inline_script = "tech_weights/avb_orbital_weight"
		modifier = {
			factor = 1.25
			has_trait_in_council = { TRAIT = leader_trait_custom_AI_assistant }
		}
	}
	
	ai_weight = {
		weight = 2
		modifier = {
			factor = 2
			OR = {
				has_ethic = ethic_materialist
				has_ethic = ethic_fanatic_materialist
				is_machine_empire = yes
			}
		}
		modifier = {
			factor = 0
			NOR = {
				has_ai_personality_behaviour = robot_exploiter
				has_ai_personality_behaviour = robot_liberator
			}
		}
		modifier = {
			factor = 2
			OR = {
				has_origin = origin_mechanists
				has_origin = origin_machine
			}
		}
		inline_script = "tech_weights/avb_orbital_weight"
		modifier = {
			factor = 1.25
			has_trait_in_council = { TRAIT = leader_trait_custom_AI_assistant }
		}
	}
}

	weight =  @tier4weight1

		exists = owner
		always = no
	}

	weight_modifier = {
